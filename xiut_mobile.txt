local function createButton(parent, label, onClick)
    local button = Instance.new("TextButton")
    button.Size = UDim2.new(1, 0, 0, 40)
    button.BackgroundColor3 = Color3.fromRGB(30, 30, 36)
    button.Text = label
    button.TextColor3 = Color3.fromRGB(200, 200, 200)
    button.Font = Enum.Font.GothamMedium
    button.TextSize = 14
    button.Parent = parent
    
    addCorner(button, 8)
    addStroke(button, Color3.fromRGB(48, 48, 56))

    -- Hover / Touch effect
    button.MouseEnter:Connect(function()
        button.BackgroundColor3 = Color3.fromRGB(40, 40, 48)
    end)
    button.MouseLeave:Connect(function()
        button.BackgroundColor3 = Color3.fromRGB(30, 30, 36)
    end)

    button.MouseButton1Click:Connect(onClick)
    return button
end

local function updateButtonState(btn, state, activeLabel, inactiveLabel)
    if state then
        btn.Text = activeLabel
        btn.TextColor3 = Color3.fromRGB(0, 255, 180)
        btn.UIStroke.Color = Color3.fromRGB(0, 180, 120)
    else
        btn.Text = inactiveLabel
        btn.TextColor3 = Color3.fromRGB(200, 200, 200)
        btn.UIStroke.Color = Color3.fromRGB(48, 48, 56)
    end
end

-- Draggable Logic (Multiplatform Touch + Mouse)
local function makeDraggable(guiFrame)
    local dragging, dragInput, dragStart, startPosition

    local function update(input)
        local delta = input.Position - dragStart
        guiFrame.Position = UDim2.new(
            startPosition.X.Scale, 
            startPosition.X.Offset + delta.X, 
            startPosition.Y.Scale, 
            startPosition.Y.Offset + delta.Y
        )
    end

    guiFrame.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPosition = guiFrame.Position

            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)

    guiFrame.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if input == dragInput and dragging then
            update(input)
        end
    end)
end

makeDraggable(mainFrame)
makeDraggable(floatButton)

-- Event-driven ESP and Hitbox Processor (Optimized to change values only when needed)
local function clearEspIndicators(character)
    if not character then return end
    local highlight = character:FindFirstChild("XiutHighlight")
    if highlight then highlight:Destroy() end

    local billboard = character:FindFirstChild("XiutBillboard")
    if billboard then billboard:Destroy() end
end

local function processPlayer(p)
    if p == localPlayer then return end
    
    local char = p.Character
    if not char then return end
    
    local root = char:FindFirstChild("HumanoidRootPart")
    local head = char:FindFirstChild("Head")
    
    -- Cache status checks
    local onSameTeam = p.Team == localPlayer.Team
    local shouldApply = not (settings.ignoreTeam and onSameTeam)
    
    -- 1. Hitbox expansion (only assign when size or state changes)
    if root then
        if settings.hitboxEnabled and shouldApply then
            local targetSize = Vector3.new(settings.hitboxSize, settings.hitboxSize, settings.hitboxSize)
            if root.Size ~= targetSize then
                root.Size = targetSize
                root.CanCollide = false
                root.Transparency = settings.transparency
                root.Color = Color3.fromRGB(255, 0, 50)
                root.Material = Enum.Material.Neon
            end
        else
            -- Reset hitbox size back to standard Roblox values if not enabled
            local defaultSize = Vector3.new(2, 2, 1)
            if root.Size ~= defaultSize then
                root.Size = defaultSize
                root.CanCollide = true
                root.Transparency = 1
            end
        end
    end

    -- 2. ESP Glow (Highlight) - Reuse highlight instead of rebuilding
    if settings.espEnabled and shouldApply then
        local highlight = char:FindFirstChild("XiutHighlight")
        if not highlight then
            highlight = Instance.new("Highlight")
            highlight.Name = "XiutHighlight"
            highlight.OutlineColor = Color3.fromRGB(255, 255, 255)
            highlight.FillTransparency = 0.5
            highlight.OutlineTransparency = 0.1
            highlight.Parent = char
        end
        local targetColor = onSameTeam and Color3.fromRGB(0, 255, 0) or Color3.fromRGB(255, 0, 0)
        if highlight.FillColor ~= targetColor then
            highlight.FillColor = targetColor
        end
    else
        local highlight = char:FindFirstChild("XiutHighlight")
        if highlight then highlight:Destroy() end
    end

    -- 3. ESP Names (Billboard)
    if settings.espNamesEnabled and shouldApply and head then
        local billboard = char:FindFirstChild("XiutBillboard")
        if not billboard then
            billboard = Instance.new("BillboardGui")
            billboard.Name = "XiutBillboard"
            billboard.AlwaysOnTop = true
            billboard.Size = UDim2.fromOffset(120, 25)
            billboard.StudsOffset = Vector3.new(0, 2.5, 0)
            
            local nameLabel = Instance.new("TextLabel")
            nameLabel.Size = UDim2.fromScale(1, 1)
            nameLabel.BackgroundTransparency = 1
            nameLabel.TextColor3 = onSameTeam and Color3.fromRGB(100, 255, 100) or Color3.fromRGB(255, 100, 100)
            nameLabel.Font = Enum.Font.GothamBold
            nameLabel.TextSize = 11
            nameLabel.TextOutlineTransparency = 0
            nameLabel.TextOutlineColor3 = Color3.new(0,0,0)
            nameLabel.Parent = billboard
            
            billboard.Parent = char
        end
        
        -- Add to list for light-weight distance tracking
        activeESPObjects[p.UserId] = billboard
    else
        local billboard = char:FindFirstChild("XiutBillboard")
        if billboard then billboard:Destroy() end
        activeESPObjects[p.UserId] = nil
    end
end

updateAllPlayers = function()
    for _, p in ipairs(Players:GetPlayers()) do
        pcall(function()
            processPlayer(p)
        end)
    end
end

-- Handle character spawn/respawn connections
local function setupPlayerConnections(p)
    if p == localPlayer then return end
    
    local function onCharAdded(char)
        task.wait(0.5) -- Wait briefly for character parts to initialize safely
        pcall(function()
            processPlayer(p)
        end)
    end

    local function onCharRemoving(char)
        pcall(function()
            clearEspIndicators(char)
            activeESPObjects[p.UserId] = nil
        end)
    end

    -- Clean up previous connections if any
    if playerConnections[p.UserId] then
        for _, conn in ipairs(playerConnections[p.UserId]) do
            conn:Disconnect()
        end
    end

    local conns = {}
    table.insert(conns, p.CharacterAdded:Connect(onCharAdded))
    table.insert(conns, p.CharacterRemoving:Connect(onCharRemoving))
    playerConnections[p.UserId] = conns

    if p.Character then
        onCharAdded(p.Character)
    end
end

local function removePlayerConnections(p)
    pcall(function()
        if p.Character then
            clearEspIndicators(p.Character)
        end
    end)
    activeESPObjects[p.UserId] = nil
    if playerConnections[p.UserId] then
        for _, conn in ipairs(playerConnections[p.UserId]) do
            conn:Disconnect()
        end
        playerConnections[p.UserId] = nil
    end
end

-- Initialize for existing players
for _, p in ipairs(Players:GetPlayers()) do
    setupPlayerConnections(p)
end

-- Listen for new players entering or leaving
Players.PlayerAdded:Connect(setupPlayerConnections)
Players.PlayerRemoving:Connect(removePlayerConnections)

-- Lightweight distance update loop (Runs every 0.5s ONLY for updating distance label)
task.spawn(function()
    while true do
        task.wait(0.5)
        
        pcall(function()
            local myChar = localPlayer.Character
            local myRoot = myChar and myChar:FindFirstChild("HumanoidRootPart")
            
            if myRoot and settings.espNamesEnabled then
                local myPos = myRoot.Position
                for userId, billboard in pairs(activeESPObjects) do
                    local p = Players:GetPlayerByUserId(userId)
                    if p and p.Character and billboard and billboard.Parent then
                        local root = p.Character:FindFirstChild("HumanoidRootPart")
                        local label = billboard:FindFirstChildOfClass("TextLabel")
                        if root and label then
                            local dist = math.floor((myPos - root.Position).Magnitude)
                            label.Text = string.format("%s [%dm]", p.Name, dist)
                        end
                    else
                        activeESPObjects[userId] = nil
                    end
                end
            end
        end)
    end
end)

-- Cleanup function to fully remove the script effects when closing/destroying ScreenGui
screenGui.Destroying:Connect(function()
    -- Disconnect all events
    for userId, conns in pairs(playerConnections) do
        for _, conn in ipairs(conns) do
            conn:Disconnect()
        end
    end
    -- Reset all hitboxes and clear indicators
    for _, p in ipairs(Players:GetPlayers()) do
        pcall(function()
            if p.Character then
                clearEspIndicators(p.Character)
                local root = p.Character:FindFirstChild("HumanoidRootPart")
                if root then
                    root.Size = Vector3.new(2, 2, 1)
                    root.CanCollide = true
                    root.Transparency = 1
                end
            end
        end)
    end
end)
