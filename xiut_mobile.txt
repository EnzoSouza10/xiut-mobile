local function addSlider(text, min, max, default, onChange)
	local f = Instance.new("Frame"); f.Size = UDim2.new(1, 0, 0, 52)
	f.BackgroundColor3 = theme.card; f.Parent = scroll; addCorner(f, 8); addStroke(f)

	local lbl = Instance.new("TextLabel"); lbl.Size = UDim2.new(0.65, 0, 0, 20)
	lbl.Position = UDim2.fromOffset(12, 5); lbl.BackgroundTransparency = 1
	lbl.Text = text; lbl.TextColor3 = theme.text; lbl.Font = Enum.Font.GothamMedium
	lbl.TextSize = 12; lbl.TextXAlignment = Enum.TextXAlignment.Left; lbl.Parent = f

	local valLbl = Instance.new("TextLabel"); valLbl.Size = UDim2.new(0.3, 0, 0, 20)
	valLbl.Position = UDim2.new(0.7, 0, 0, 5); valLbl.BackgroundTransparency = 1
	valLbl.Text = tostring(default); valLbl.TextColor3 = theme.blue; valLbl.Font = Enum.Font.GothamBold
	valLbl.TextSize = 13; valLbl.TextXAlignment = Enum.TextXAlignment.Right; valLbl.TextScaled = true; valLbl.Parent = f

	local track = Instance.new("Frame"); track.Size = UDim2.new(1, -24, 0, 6)
	track.Position = UDim2.new(0, 12, 0, 32); track.BackgroundColor3 = Color3.fromRGB(50, 50, 60)
	track.BorderSizePixel = 0; track.Parent = f; addCorner(track, 3)

	local fill = Instance.new("Frame"); fill.Size = UDim2.fromScale((default - min) / (max - min), 1)
	fill.BackgroundColor3 = theme.blue; fill.BorderSizePixel = 0; fill.Parent = track; addCorner(fill, 3)

	local thumb = Instance.new("TextButton"); thumb.Size = UDim2.fromOffset(16, 16)
	thumb.Position = UDim2.new((default - min) / (max - min), -8, 0.5, -8)
	thumb.BackgroundColor3 = Color3.fromRGB(255, 255, 255); thumb.Text = ""; thumb.Parent = track
	addCorner(thumb, 8); addStroke(thumb, theme.blue, 1.5)

	local dragging = false
	local function update(inp)
		local w = track.AbsoluteSize.X; if w <= 0 then return end
		local ratio = math.clamp((inp.Position.X - track.AbsolutePosition.X) / w, 0, 1)
		fill.Size = UDim2.fromScale(ratio, 1); thumb.Position = UDim2.new(ratio, -8, 0.5, -8)
		local val = min + (max - min) * ratio; valLbl.Text = tostring(math.floor(val * 100) / 100); onChange(val)
	end

	thumb.InputBegan:Connect(function(i)
		if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then dragging = true end
	end)
	local c1 = UserInputService.InputChanged:Connect(function(i)
		if dragging and (i.UserInputType == Enum.UserInputType.MouseMovement or i.UserInputType == Enum.UserInputType.Touch) then update(i) end
	end)
	local c2 = UserInputService.InputEnded:Connect(function(i)
		if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then dragging = false end
	end)
	f.Destroying:Connect(function() c1:Disconnect(); c2:Disconnect() end)
	return { set = function(val)
		local r = (val - min) / (max - min); fill.Size = UDim2.fromScale(r, 1)
		thumb.Position = UDim2.new(r, -8, 0.5, -8); valLbl.Text = tostring(math.floor(val * 100) / 100)
	end }
end

-- Draggable
local function makeDraggable(g)
	local drag, inp, start, pos
	local function up(i) local d = i.Position - start; g.Position = UDim2.new(pos.X.Scale, pos.X.Offset + d.X, pos.Y.Scale, pos.Y.Offset + d.Y) end
	g.InputBegan:Connect(function(i)
		if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then
			drag = true; start = i.Position; pos = g.Position
			i.Changed:Connect(function() if i.UserInputState == Enum.UserInputState.End then drag = false end end)
		end
	end)
	g.InputChanged:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseMovement or i.UserInputType == Enum.UserInputType.Touch then inp = i end end)
	UserInputService.InputChanged:Connect(function(i) if i == inp and drag then up(i) end end)
end
makeDraggable(mainFrame); makeDraggable(floatBtn)

-- === HITBOX ENGINE (Direct Property Approach) ===

local playerConns = {}
local espBillboards = {}

local function clearESP(char)
	if not char then return end
	local h = char:FindFirstChild("XiutHighlight"); if h then h:Destroy() end
	local b = char:FindFirstChild("XiutBillboard"); if b then b:Destroy() end
end

local defaultSizes = {
	HumanoidRootPart = Vector3.new(2, 2, 1),
	Torso = Vector3.new(2, 2, 1),
	Head = Vector3.new(1, 1, 1),
}

local function resetPart(part)
	if not part then return end
	if defaultSizes[part.Name] then
		part.Size = defaultSizes[part.Name]
	end
	part.CanCollide = true
	part.Transparency = 1
	part.Material = Enum.Material.Plastic
end

local function expandPart(part)
	if not part then return end
	local size = config.hitboxSize
	part.Size = Vector3.new(size, size, size)
	part.CanCollide = not config.noCollide
	part.Transparency = config.transparency
	part.Material = Enum.Material.Neon
	part.Color = Color3.fromRGB(255, 0, 50)
end

local function processPlayer(p)
	if p == localPlayer then return end

	local char = p.Character
	if not char then return end

	local onSameTeam = p.Team == localPlayer.Team
	local shouldApply = not (config.ignoreTeam and onSameTeam)

	-- Hitbox: expand ALL body parts, not just HRP
	if config.hitboxEnabled and shouldApply then
		for _, part in pairs(char:GetChildren()) do
			if part:IsA("BasePart") then
				expandPart(part)
			end
		end
	else
		for _, part in pairs(char:GetChildren()) do
			if part:IsA("BasePart") then
				resetPart(part)
			end
		end
	end

	-- ESP Glow
	if config.espEnabled and shouldApply then
		local hl = char:FindFirstChild("XiutHighlight")
		if not hl then
			hl = Instance.new("Highlight"); hl.Name = "XiutHighlight"
			hl.OutlineColor = Color3.fromRGB(255, 255, 255); hl.FillTransparency = 0.5; hl.OutlineTransparency = 0.1
			hl.Parent = char
		end
		hl.FillColor = onSameTeam and Color3.fromRGB(0, 255, 0) or Color3.fromRGB(255, 0, 0)
	else
		local hl = char:FindFirstChild("XiutHighlight"); if hl then hl:Destroy() end
	end

	-- ESP Name
	local head = char:FindFirstChild("Head")
	if config.espNamesEnabled and shouldApply and head then
		local bb = char:FindFirstChild("XiutBillboard")
		if not bb then
			bb = Instance.new("BillboardGui"); bb.Name = "XiutBillboard"
			bb.AlwaysOnTop = true; bb.Size = UDim2.fromOffset(120, 25); bb.StudsOffset = Vector3.new(0, 2.5, 0)
			local lbl = Instance.new("TextLabel"); lbl.Size = UDim2.fromScale(1, 1); lbl.BackgroundTransparency = 1
			lbl.TextColor3 = onSameTeam and Color3.fromRGB(100, 255, 100) or Color3.fromRGB(255, 100, 100)
			lbl.Font = Enum.Font.GothamBold; lbl.TextSize = 11
			lbl.TextStrokeTransparency = 0; lbl.TextStrokeColor3 = Color3.new(0, 0, 0)
			lbl.Parent = bb; bb.Parent = char
		end
		espBillboards[p.UserId] = bb
	else
		local bb = char:FindFirstChild("XiutBillboard"); if bb then bb:Destroy() end
		espBillboards[p.UserId] = nil
	end
end

local function updateAll()
	for _, p in ipairs(Players:GetPlayers()) do
		pcall(function() processPlayer(p) end)
	end
end

-- Player connections
local function setupConns(p)
	if p == localPlayer then return end
	if playerConns[p.UserId] then
		for _, c in ipairs(playerConns[p.UserId]) do c:Disconnect() end
	end

	local conns = {}
	table.insert(conns, p.CharacterAdded:Connect(function(char)
		task.wait(0.5); pcall(function() processPlayer(p) end)
	end))
	table.insert(conns, p.CharacterRemoving:Connect(function(char)
		pcall(function() clearESP(char); espBillboards[p.UserId] = nil end)
	end))
	playerConns[p.UserId] = conns

	if p.Character then
		task.wait(0.5); pcall(function() processPlayer(p) end)
	end
end

local function removeConns(p)
	pcall(function() if p.Character then clearESP(p.Character) end end)
	espBillboards[p.UserId] = nil
	if playerConns[p.UserId] then
		for _, c in ipairs(playerConns[p.UserId]) do c:Disconnect() end
		playerConns[p.UserId] = nil
	end
end

for _, p in ipairs(Players:GetPlayers()) do setupConns(p) end
Players.PlayerAdded:Connect(setupConns)
Players.PlayerRemoving:Connect(removeConns)

-- Distance loop
task.spawn(function()
	while true do
		task.wait(0.5)
		pcall(function()
			local myChar = localPlayer.Character; local myRoot = myChar and myChar:FindFirstChild("HumanoidRootPart")
			if myRoot and config.espNamesEnabled then
				local myPos = myRoot.Position
				for uid, bb in pairs(espBillboards) do
					local p = Players:GetPlayerByUserId(uid)
					if p and p.Character and bb and bb.Parent then
						local root = p.Character:FindFirstChild("HumanoidRootPart")
						local lbl = bb:FindFirstChildOfClass("TextLabel")
						if root and lbl then
							local dist = math.floor((myPos - root.Position).Magnitude)
							lbl.Text = p.Name .. " [" .. dist .. "m]"
						end
					else
						espBillboards[uid] = nil
					end
				end
			end
		end)
	end
end)

-- Cleanup
screenGui.Destroying:Connect(function()
	for uid, conns in pairs(playerConns) do
		for _, c in ipairs(conns) do c:Disconnect() end
	end
	for _, p in ipairs(Players:GetPlayers()) do
		pcall(function()
			if p.Character then
				clearESP(p.Character)
				for _, part in pairs(p.Character:GetChildren()) do
					if part:IsA("BasePart") then resetPart(part) end
				end
			end
		end)
	end
end)
