local function criarSlider(txt, min, max, default, callback)
	local bg = Instance.new("Frame")
	bg.Size = UDim2.new(0, 220, 0, 45)
	bg.Position = UDim2.new(0.5, -110, 0, yPos)
	bg.BackgroundColor3 = Color3.fromRGB(30, 30, 45)
	bg.BorderSizePixel = 0
	bg.Parent = scrollFrame

	local label = Instance.new("TextLabel")
	label.Size = UDim2.new(1, -10, 0, 18)
	label.Position = UDim2.new(0, 5, 0, 2)
	label.BackgroundTransparency = 1
	label.TextColor3 = Color3.fromRGB(180, 180, 220)
	label.Text = txt .. ": " .. tostring(default)
	label.Font = Enum.Font.Gotham
	label.TextSize = 12
	label.TextXAlignment = Enum.TextXAlignment.Left
	label.Parent = bg

	local barraBg = Instance.new("Frame")
	barraBg.Size = UDim2.new(1, -10, 0, 8)
	barraBg.Position = UDim2.new(0, 5, 0, 22)
	barraBg.BackgroundColor3 = Color3.fromRGB(60, 60, 80)
	barraBg.BorderSizePixel = 0
	barraBg.Parent = bg

	local preenchimento = Instance.new("Frame")
	preenchimento.Size = UDim2.new((default - min) / (max - min), 0, 1, 0)
	preenchimento.BackgroundColor3 = Color3.fromRGB(100, 150, 255)
	preenchimento.BorderSizePixel = 0
	preenchimento.Parent = barraBg

	local valorAtual = default
	local dragging = false

	local function atualizar(input)
		local pos = math.clamp((input.Position.X - barraBg.AbsolutePosition.X) / barraBg.AbsoluteSize.X, 0, 1)
		valorAtual = min + math.floor(pos * (max - min) + 0.5)
		preenchimento.Size = UDim2.new(pos, 0, 1, 0)
		label.Text = txt .. ": " .. tostring(valorAtual)
		if callback then callback(valorAtual) end
	end

	barraBg.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = true
			atualizar(input)
		end
	end)

	barraBg.InputEnded:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = false
		end
	end)

	UserInputService.InputChanged:Connect(function(input)
		if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
			atualizar(input)
		end
	end)

	yPos = yPos + 50
	return { GetValue = function() return valorAtual end }
end

local function parteAtiva(partName)
	for _, nome in ipairs(listaPartes) do
		local t = togglePartes[nome]
		if t and t.GetValue() then
			local match = false
			if string.match(partName, nome) then match = true
			elseif nome == "Custom Part" and string.match(partName, customPartName) then match = true
			elseif nome == "Left Arm" and string.match(partName, "Left") and (string.match(partName, "Arm") or string.match(partName, "Hand")) then match = true
			elseif nome == "Right Arm" and string.match(partName, "Right") and (string.match(partName, "Arm") or string.match(partName, "Hand")) then match = true
			elseif nome == "Left Leg" and string.match(partName, "Left") and (string.match(partName, "Leg") or string.match(partName, "Foot")) then match = true
			elseif nome == "Right Leg" and string.match(partName, "Right") and (string.match(partName, "Leg") or string.match(partName, "Foot")) then match = true
			end
			if match then return true end
		end
	end
	return false
end

local function getTamanho(partName)
	for _, nome in ipairs(listaPartes) do
		local t = togglePartes[nome]
		if t and t.GetValue() then
			local match = false
			if string.match(partName, nome) then match = true
			elseif nome == "Custom Part" and string.match(partName, customPartName) then match = true
			elseif nome == "Left Arm" and string.match(partName, "Left") and (string.match(partName, "Arm") or string.match(partName, "Hand")) then match = true
			elseif nome == "Right Arm" and string.match(partName, "Right") and (string.match(partName, "Arm") or string.match(partName, "Hand")) then match = true
			elseif nome == "Left Leg" and string.match(partName, "Left") and (string.match(partName, "Leg") or string.match(partName, "Foot")) then match = true
			elseif nome == "Right Leg" and string.match(partName, "Right") and (string.match(partName, "Leg") or string.match(partName, "Foot")) then match = true
			end
			if match then return tamanhos[nome] end
		end
	end
	return nil
end

local function isDead(char)
	if not char then return true end
	local h = char:FindFirstChildWhichIsA("Humanoid")
	return h and h.Health <= 0
end

local function isSitting(char)
	if not ignorarSentados then return false end
	local h = char:FindFirstChildWhichIsA("Humanoid")
	return h and h.Sit == true
end

local function isFFed(char)
	if not ignorarFF or not char then return false end
	local ff = char:FindFirstChildWhichIsA("ForceField")
	return ff ~= nil and ff.Visible == true
end

local function addPlayer(player)
	if player == lPlayer then return end
	players[player] = {}
	local idx = players[player]
	local char = player.Character
	local defaultProps = {}

	local nameEsp = Drawing.new("Text")
	nameEsp.Center = true
	nameEsp.Outline = true
	nameEsp.Visible = false

	local chams = Instance.new("Highlight")
	chams.Enabled = false
	chams.Parent = game:GetService("CoreGui")

	local function resize(char)
		if not char then return end
		for _, v in pairs(char:GetChildren()) do
			if v:IsA("BasePart") then
				if not defaultProps[v.Name] then
					defaultProps[v.Name] = {
						Size = v.Size,
						Transparency = v.Transparency,
						Massless = v.Massless,
						CanCollide = v.CanCollide
					}
				end
				if enabled and parteAtiva(v.Name) and not isDead(char) and not isSitting(char) and not isFFed(char) then
					local tam = getTamanho(v.Name) or 10
					v.Size = Vector3.new(tam, tam, tam)
					v.Transparency = transparencia
					v.Massless = true
					v.CanCollide = false
					if v.Name == "Head" then
						local face = v:FindFirstChild("face")
						if face then face.Transparency = transparencia end
					end
				else
					local props = defaultProps[v.Name]
					if props then
						v.Size = props.Size
						v.Transparency = props.Transparency
						v.Massless = props.Massless
						v.CanCollide = props.CanCollide
						if v.Name == "Head" then
							local face = v:FindFirstChild("face")
							if face then face.Transparency = props.Transparency end
						end
					end
				end
			end
		end
	end

	function idx:Update()
		resize(char)
	end

	function idx:UpdateESP()
		Camera = Workspace.CurrentCamera
		if not char or isDead(char) then
			nameEsp.Visible = false
			chams.Enabled = false
			return
		end
		if nomesVisiveis then
			local torso = char:FindFirstChild("Torso") or char:FindFirstChild("HumanoidRootPart") or char:FindFirstChild("UpperTorso")
			if torso then
				local pos, vis = Camera:WorldToViewportPoint(torso.Position)
				if vis then
					nameEsp.Text = player.DisplayName or player.Name
					nameEsp.Color = player.TeamColor.Color
					nameEsp.Position = Vector2.new(pos.X, pos.Y)
					nameEsp.Size = 1000 / pos.Z + 10
					nameEsp.Visible = true
				else
					nameEsp.Visible = false
				end
			else
				nameEsp.Visible = false
			end
		else
			nameEsp.Visible = false
		end
		if chamsVisiveis then
			chams.Adornee = char
			chams.FillColor = player.TeamColor.Color
			chams.OutlineColor = Color3.new(0, 0, 0)
			chams.FillTransparency = 0.5
			chams.Enabled = true
		else
			chams.Enabled = false
		end
	end

	function idx:DeleteVisuals()
		nameEsp:Remove()
		chams:Destroy()
	end

	if char then
		task.wait(0.5)
		idx:Update()
	end

	player.CharacterAdded:Connect(function(c)
		char = c
		defaultProps = {}
		task.wait(0.5)
		idx:Update()
	end)
	player.CharacterRemoving:Connect(function()
		char = nil
		defaultProps = {}
	end)
end

for _, p in ipairs(Players:GetPlayers()) do
	addPlayer(p)
end

		players[player]:DeleteVisuals()
		players[player] = nil
	end

			idx:Update()
			idx:UpdateESP()
		end)
	end
