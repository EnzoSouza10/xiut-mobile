-- Config
local cfg = {en=false,sz=8,tr=0.5,nc=false,es=false,ig=true,ad=true,adD=0.12}
local sel = {}
for _,n in pairs({"HumanoidRootPart","Head","Torso","Left Arm","Right Arm","Left Leg","Right Leg"}) do sel[n]=true end

-- Player data
local pd = {}
local df = {}
local drw = {}
local hb
local adActive = false

local function isTeam(p)
	if not p.Character then return false end
	if not lp.Character then return false end
	return lp.Team == p.Team
end

local function sv(c)
	if not c then return end
	df[c] = {}
	for _,p in pairs(c:GetChildren()) do
		if p:IsA("BasePart") then
			df[c][p] = {Size=p.Size,T=p.Transparency,CC=p.CanCollide,ML=p.Massless}
		end
	end
end

local function rs(char)
	if not char or not df[char] then return end
	for p,pr in pairs(df[char]) do
		if p and p.Parent then
			p.Size=pr.Size;p.Transparency=pr.T;p.CanCollide=pr.CC;p.Massless=pr.ML
		end
	end
end

local function isPartSelected(name)
	if sel["HumanoidRootPart"] and name == "HumanoidRootPart" then return true end
	if sel["Head"] and (name == "Head" or name:find("Head")) then return true end
	if sel["Torso"] and (name:find("Torso") or name:find("Torso")) then return true end
	if sel["Left Arm"] and name:find("Left") and (name:find("Arm") or name:find("Hand") or name:find("Forearm") or name:find("Shoulder")) then return true end
	if sel["Right Arm"] and name:find("Right") and (name:find("Arm") or name:find("Hand") or name:find("Forearm") or name:find("Shoulder")) then return true end
	if sel["Left Leg"] and name:find("Left") and (name:find("Leg") or name:find("Foot") or name:find("Thigh") or name:find("Calf")) then return true end
	if sel["Right Leg"] and name:find("Right") and (name:find("Leg") or name:find("Foot") or name:find("Thigh") or name:find("Calf")) then return true end
	return false
end

local function ex(char, plr)
	if not char then return end
	if plr and cfg.ig and isTeam(plr) then rs(char); return end
	for _,p in pairs(char:GetChildren()) do
		if p:IsA("BasePart") and isPartSelected(p.Name) then
			p.Size = Vector3.new(cfg.sz,cfg.sz,cfg.sz)
			p.Transparency = cfg.tr
			p.CanCollide = not cfg.nc
			if p.Name ~= "HumanoidRootPart" then p.Massless = true end
		end
	end
end

local function ad()
	if not cfg.ad or not cfg.en or adActive then return end
	adActive = true
	for p,pl in pairs(pd) do
		if pl.al and p.Character and not (cfg.ig and isTeam(p)) then
			rs(p.Character)
		end
	end
	spawnThread(function() wf(cfg.adD) adActive = false end)
end

local function ap(p)
	if p == lp then return end
	local pl = {al=true,ig=false,ch=nil}
	pd[p] = pl
	local function oc(char)
		if not char then return end
		pl.al = true; pl.ch = char
		local h = char:FindFirstChildWhichIsA("Humanoid")
		if h then
			h.Died:Connect(function() pl.al = false end)
		end
		local root = char:WaitForChild("HumanoidRootPart", 5)
		if not root then root = char:WaitForChild("Torso", 3) end
		if not root then return end
		sv(char)
		if cfg.en then ex(char, p) end
		char.ChildAdded:Connect(function(ch) wf() if ch:IsA("BasePart") and cfg.en then ex(char, p) end end)
	end
	if p.Character then oc(p.Character) end
	p.CharacterAdded:Connect(function(char) wf(0.5) oc(char) end)
	p.CharacterRemoving:Connect(function() pl.al = false end)
end
for _,p in pairs(P:GetPlayers()) do ap(p) end
P.PlayerAdded:Connect(ap)
P.PlayerRemoving:Connect(function(p) pd[p] = nil end)

-- Heartbeat enforcement loop
hb = R.Heartbeat:Connect(function()
	if not cfg.en or adActive then return end
	for p,pl in pairs(pd) do
		if pl.al and p.Character and not (cfg.ig and isTeam(p)) then
			ex(p.Character, p)
		end
	end
end)

-- ESP
R.RenderStepped:Connect(function()
	if not cfg.es then
		for _,d in pairs(drw) do d.Visible = false end
		return
	end
	C = W.CurrentCamera
	if not C then return end
	for p,pl in pairs(pd) do
		local ch = p.Character
		if not ch or not pl.al or (cfg.ig and isTeam(p)) then
			if drw[p] then drw[p].Visible = false end
		else
			local rt = ch:FindFirstChild("HumanoidRootPart") or ch:FindFirstChild("Torso")
			if rt then
				if not drw[p] then
					local s,d = pcall(function()
						local d = Drawing.new("Text"); d.Center = true; d.Outline = true; d.Size = 14; d.OutlineColor = Color3.fromRGB(0,0,0); return d
					end)
					if not s then return end
					drw[p] = d
				end
				local pos, vis = C:WorldToViewportPoint(rt.Position)
				if vis then
					local ds = math.floor((C.CFrame.Position-rt.Position).Magnitude)
					drw[p].Color = p.Team == lp.Team and Color3.fromRGB(100,255,100) or Color3.fromRGB(255,100,100)
					drw[p].Text = p.Name.." ["..ds.."m]"
					drw[p].Position = Vector2.new(pos.X,pos.Y-45)
					drw[p].Visible = true
				else
					drw[p].Visible = false
				end
			elseif drw[p] then
				drw[p].Visible = false
			end
		end
	end
end)

		ad()
	end
end)

	wf(0.5)
	local t = char:FindFirstChildWhichIsA("Tool")
	if t then t.Activated:Connect(function() if cfg.ad and cfg.en then ad() end end) end
	char.ChildAdded:Connect(function(c) wf() if c:IsA("Tool") then c.Activated:Connect(function() if cfg.ad and cfg.en then ad() end end) end end)
end)
if lp.Character then
	for _,t in pairs(lp.Character:GetChildren()) do
		if t:IsA("Tool") then
			t.Activated:Connect(function() if cfg.ad and cfg.en then ad() end end)
		end
	end
end
