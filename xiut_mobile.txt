-- === HITBOX ENGINE ===
-- Try MT-Api first, fallback to direct property setting
local mtApiMode = false

local function tryLoadMTAPI()
	local ok = false
	pcall(function() if env.MTAPIMutex then ok = true end end)
	if ok then return true end

	local url = "https://raw.githubusercontent.com/RectangularObject/MT-Api-v2/main/__source/mt-api%20v2.lua"
	pcall(function()
		local code = game:HttpGet(url, true)
		loadstring(code)()
	end)
	pcall(function() if env.MTAPIMutex then ok = true end end)
	if ok then return true end

	pcall(function()
		local code = game:HttpGetAsync(url)
		loadstring(code)()
	end)
	pcall(function() if env.MTAPIMutex then ok = true end end)
	if ok then return true end

	return false
end

mtApiMode = tryLoadMTAPI()

-- Player data
local players = {}
local espDrawings = {}

local function isTeammate(p, pc)
	if not pc then return true end
	if not lp.Character then return false end
	return lp.Team == p.Team
end

local function isDead(pc)
	if not pc then return true end
	local hum = pc:FindFirstChildWhichIsA("Humanoid")
	if hum then return hum.Health <= 0 end
	return true
end

-- Direct property approach (fallback)
local defaultProps = {}
local function saveDefaults(char)
	if not char then return end
	defaultProps[char] = {}
	for _, part in pairs(char:GetChildren()) do
		if part:IsA("BasePart") then
			defaultProps[char][part] = { Size = part.Size, Transparency = part.Transparency, CanCollide = part.CanCollide, Massless = part.Massless }
		end
	end
end

local function resetParts(char)
	if not char or not defaultProps[char] then return end
	for part, props in pairs(defaultProps[char]) do
		if part and part.Parent then
			part.Size = props.Size; part.Transparency = props.Transparency; part.CanCollide = props.CanCollide; part.Massless = props.Massless
		end
	end
end

local function expandParts(char)
	if not char then return end
	local ignored = config.ignoreTeam and players[char] and players[char].ignored
	if ignored then
		resetParts(char)
		return
	end
	for _, part in pairs(char:GetChildren()) do
		if part:IsA("BasePart") and selectedParts[part.Name] then
			local size = config.size
			part.Size = Vector3.new(size, size, size)
			part.Transparency = config.transparency
			part.CanCollide = not config.noCollide
			if part.Name ~= "HumanoidRootPart" then
				part.Massless = true
			end
		end
	end
end

-- MT-Api approach
local function hookPart(part, pData)
	if not part:IsA("BasePart") then return end
	local props = { Size = part.Size, Transparency = part.Transparency, Massless = part.Massless, CanCollide = part.CanCollide }
	local ok = pcall(function()
		local gs = part:AddGetHook("Size", props.Size)
		local gt = part:AddGetHook("Transparency", props.Transparency)
		local gm = part:AddGetHook("Massless", props.Massless)
		local gc = part:AddGetHook("CanCollide", props.CanCollide)
		local ss = part:AddSetHook("Size", function(_, v)
			props.Size = v; gs:Modify("Size", props.Size)
			if config.enabled and pData.alive and not pData.ignored and selectedParts[part.Name] then return Vector3.new(config.size, config.size, config.size) end
			return props.Size
		end)
		local st = part:AddSetHook("Transparency", function(_, v)
			props.Transparency = v; gt:Modify("Transparency", props.Transparency)
			if config.enabled and pData.alive and not pData.ignored and selectedParts[part.Name] then return config.transparency end
			return props.Transparency
		end)
		local sm = part:AddSetHook("Massless", function(_, v)
			props.Massless = v; gm:Modify("Massless", props.Massless)
			if config.enabled and part.Name ~= "HumanoidRootPart" and pData.alive and not pData.ignored and selectedParts[part.Name] then return true end
			return props.Massless
		end)
		local sc = part:AddSetHook("CanCollide", function(_, v)
			props.CanCollide = v; gc:Modify("CanCollide", props.CanCollide)
			if config.enabled and config.noCollide and pData.alive and not pData.ignored and selectedParts[part.Name] then return false end
			return props.CanCollide
		end)
		part.Destroying:Connect(function()
			gs:Remove(); gt:Remove(); gm:Remove(); gc:Remove()
			ss:Remove(); st:Remove(); sm:Remove(); sc:Remove()
		end)
	end)
	if not ok then
		mtApiMode = false
	end
end

local function setupPlayer(p)
	if p == lp or players[p] then return end
	local pData = { alive = true, ignored = false }
	players[p] = pData

	local function onChar(char)
		if not char then return end
		pData.alive = true
		local hum = char:FindFirstChildWhichIsA("Humanoid")
		if hum then
			hum.StateChanged:Connect(function(_, ns) pData.alive = ns ~= Enum.HumanoidStateType.Dead end)
			hum.Died:Connect(function() pData.alive = false end)
		end
		if mtApiMode then
			for _, part in pairs(char:GetChildren()) do hookPart(part, pData) end
			char.ChildAdded:Connect(function(child) waitFn() hookPart(child, pData) end)
		else
			saveDefaults(char)
			expandParts(char)
			char.ChildAdded:Connect(function(child) waitFn() if child:IsA("BasePart") then expandParts(char) end end)
		end
	end

	if p.Character then onChar(p.Character) end
	p.CharacterAdded:Connect(function(char) waitFn(0.5) onChar(char) end)
	p.CharacterRemoving:Connect(function() pData.alive = false end)
	p:GetPropertyChangedSignal("Team"):Connect(function()
		pData.ignored = config.ignoreTeam and isTeammate(p, p.Character)
	end)
	if p.Character then
		pData.ignored = config.ignoreTeam and isTeammate(p, p.Character)
	end
end

for _, p in pairs(Players:GetPlayers()) do setupPlayer(p) end
Players.PlayerAdded:Connect(setupPlayer)
Players.PlayerRemoving:Connect(function(p) players[p] = nil end)

-- Update all (for direct mode)
local function updateAll()
	if mtApiMode then return end
	for p, pd in pairs(players) do
		if p.Character then
			if config.enabled and pd.alive and not pd.ignored then
				expandParts(p.Character)
			else
				resetParts(p.Character)
			end
		end
	end
end

-- ESP
RunService:BindToRenderStep("XiutESP", Enum.RenderPriority.Camera.Value - 1, function()
	if not config.espNames then
		for _, d in pairs(espDrawings) do d.Visible = false end
		return
	end
	Camera = Workspace.CurrentCamera
	if not Camera then return end
	for p, pd in pairs(players) do
		local char = p.Character
		if not char or not pd.alive or pd.ignored then
			if espDrawings[p] then espDrawings[p].Visible = false end
		else
			local root = char:FindFirstChild("HumanoidRootPart") or char:FindFirstChild("Torso")
			if root then
				if not espDrawings[p] then
					local s, d = pcall(function()
						local d = Drawing.new("Text"); d.Center = true; d.Outline = true; d.Size = 16; d.OutlineColor = Color3.fromRGB(0, 0, 0); return d
					end)
					if not s then return end
					espDrawings[p] = d
				end
				local pos, vis = Camera:WorldToViewportPoint(root.Position)
				if vis then
					local dist = math.floor((Camera.CFrame.Position - root.Position).Magnitude)
					espDrawings[p].Color = p.Team == lp.Team and Color3.fromRGB(100, 255, 100) or Color3.fromRGB(255, 100, 100)
					espDrawings[p].Text = p.Name .. " [" .. dist .. "m]"
					espDrawings[p].Position = Vector2.new(pos.X, pos.Y - 50)
					espDrawings[p].Visible = true
				else
					espDrawings[p].Visible = false
				end
			elseif espDrawings[p] then
				espDrawings[p].Visible = false
			end
		end
	end
end)

-- Cleanup
gui.Destroying:Connect(function()
	for _, d in pairs(espDrawings) do pcall(function() d:Remove() end) end
	if not mtApiMode then
		for p, pd in pairs(players) do if p.Character then resetParts(p.Character) end end
	end
	env.XiutLoaded = nil
end)
