local Toggle = CreateMainButton("AIMBOT OFF")

-- ==================== LOGIC ====================

-- Atualiza visual do botao toggle
local function UpdateToggle()
    if Aimbot.Enabled then
        if Aimbot.LockedTarget then
            Toggle.Text = "TRAVADO: " .. Aimbot.LockedTarget.Name
        else
            Toggle.Text = "AGUARDANDO TOQUE..."
        end

        TweenService:Create(Toggle, TweenInfo.new(0.2),
            {BackgroundColor3 = COLORS.Red}):Play()
    else
        Toggle.Text = "AIMBOT OFF"
        TweenService:Create(Toggle, TweenInfo.new(0.2),
            {BackgroundColor3 = COLORS.Black}):Play()
    end
end

-- Desativa completamente (chamado ao matar inimigo ou desligar manualmente)
local function DeactivateAimbot()
    Aimbot.Enabled      = false
    Aimbot.LockedTarget = nil
    UpdateToggle()
end

-- Busca inimigo mais proximo de uma posicao de tela (igual ao PC com cursor)
-- touchPos: Vector2 com coordenadas de tela do toque
local function GetTargetNearTouchPos(touchPos)
    local closest  = nil
    local minDist  = Aimbot.FOV

    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Character then
            local head = player.Character:FindFirstChild("Head")
            local hum  = player.Character:FindFirstChild("Humanoid")

            if head and hum and hum.Health > 0 then
                local predicted = head.Position
                    + head.AssemblyLinearVelocity * Aimbot.Prediction

                local screenPos, onScreen =
                    Camera:WorldToViewportPoint(predicted)

                if onScreen then
                    local dist =
                        (Vector2.new(screenPos.X, screenPos.Y) - touchPos).Magnitude

                    if dist < minDist then
                        minDist = dist
                        closest = player
                    end
                end
            end
        end
    end

    return closest
end

-- Toque na tela (fora da GUI): trava no inimigo mais proximo do ponto tocado
-- Usa TouchTapInWorld para ignorar toques que acertaram elementos de UI
UserInputService.TouchTapInWorld:Connect(function(touchPos, processedByUI)
    -- Se o sistema nao esta ligado, ignora
    if not Aimbot.Enabled then return end

    -- Se o toque foi consumido pela GUI, ignora
    if processedByUI then return end

    -- Ja tem alvo travado: nao substitui (precisa morrer para trocar)
    if Aimbot.LockedTarget then return end

    -- Busca inimigo proximo ao ponto tocado
    local target = GetTargetNearTouchPos(touchPos)

    if target then
        Aimbot.LockedTarget = target
        UpdateToggle()
    end
    -- Se nao achou ninguem no FOV, aguarda proximo toque
end)

-- Botao toggle: liga/desliga o sistema
Toggle.MouseButton1Click:Connect(function()
    if Aimbot.Enabled then
        -- Desliga manualmente
        DeactivateAimbot()
    else
        -- Liga: aguarda toque para travar alvo
        Aimbot.Enabled      = true
        Aimbot.LockedTarget = nil
        UpdateToggle()
    end
end)

-- ==================== SETTINGS ====================

    if n then Aimbot.FOV = n end
end)

    local n = tonumber(SmoothBox.Text)
    if n then
        Aimbot.Smoothness = math.clamp(n, 0.01, 1)
        SmoothBox.Text = tostring(Aimbot.Smoothness)
    end
end)

PredictionBox.FocusLost:Connect(function()
    local n = tonumber(PredictionBox.Text)
    if n then Aimbot.Prediction = n end
end)

-- ==================== AIMBOT LOOP ====================

RunService.RenderStepped:Connect(function()

    -- So executa se sistema ligado E com alvo travado
    if not (Aimbot.Enabled and Aimbot.LockedTarget) then return end

    local char = Aimbot.LockedTarget.Character

    -- Verifica se alvo ainda esta vivo
    if char
    and char:FindFirstChild("Head")
    and char:FindFirstChild("Humanoid")
    and char.Humanoid.Health > 0 then

        local head = char.Head

        local predicted =
            head.Position + head.AssemblyLinearVelocity * Aimbot.Prediction

        local _, onScreen = Camera:WorldToViewportPoint(predicted)

        if onScreen then
            -- Move a camera suavemente em direcao a cabeca do alvo
            -- Mesma logica de hitbox do PC, so muda o metodo de mover a camera
            Camera.CFrame = Camera.CFrame:Lerp(
                CFrame.new(Camera.CFrame.Position, predicted),
                Aimbot.Smoothness
            )
        end

    else
        -- INIMIGO MORREU OU SAIU: desativa automaticamente
        -- Precisa reativar manualmente (pressionar AIMBOT OFF -> ON)
        DeactivateAimbot()
    end
end)
