-- Draggable
local function makeDrag(g)
	local d, s, p
	g.InputBegan:Connect(function(i)
		if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then
			d = true; s = i.Position; p = g.Position
			i.Changed:Connect(function() if i.UserInputState == Enum.UserInputState.End then d = false end end)
		end
	end)
	UserInputService.InputChanged:Connect(function(i)
		if d and (i.UserInputType == Enum.UserInputType.MouseMovement or i.UserInputType == Enum.UserInputType.Touch) then
			local d = i.Position - s; g.Position = UDim2.new(p.X.Scale, p.X.Offset + d.X, p.Y.Scale, p.Y.Offset + d.Y)
		end
	end)
end

local function isPartMatch(name, sel)
	if sel == "Left Arm" then return string.match(name, "Left") and (string.match(name, "Arm") or string.match(name, "Hand")) end
	if sel == "Right Arm" then return string.match(name, "Right") and (string.match(name, "Arm") or string.match(name, "Hand")) end
	if sel == "Left Leg" then return string.match(name, "Left") and (string.match(name, "Leg") or string.match(name, "Foot")) end
	if sel == "Right Leg" then return string.match(name, "Right") and (string.match(name, "Leg") or string.match(name, "Foot")) end
	return name == sel
end

local partList = { "HumanoidRootPart", "Head", "Torso", "Left Arm", "Right Arm", "Left Leg", "Right Leg" }
local selectedParts = {}
for _, v in pairs(partList) do selectedParts[v] = true end

local function isTeammate(p, pc)
	if game.GameId == 718936923 then
		if not lp.Character or not pc or not pc:FindFirstChild("HumanoidRootPart") then return true end
		return lp.Character.HumanoidRootPart.Color == pc.HumanoidRootPart.Color
	elseif game.PlaceId == 633284182 then
		if not p:FindFirstChild("PlayerData") or not p.PlayerData:FindFirstChild("TeamValue") then return true end
		return lp.PlayerData.TeamValue.Value == p.PlayerData.TeamValue.Value
	elseif game.PlaceId == 2029250188 then
		if not lp.Character or not pc then return true end
		return lp.Character.Parent == pc.Parent
	elseif game.PlaceId == 2978450615 then
		return getrenv()._G.PlayerProfiles.Data[lp.Name].Team == getrenv()._G.PlayerProfiles.Data[p.Name].Team
	end
	return lp.Team == p.Team
end

local function isDead(pc)
	if not pc then return true end
	local hum = pc:FindFirstChildWhichIsA("Humanoid")
	return hum and hum:GetState() == Enum.HumanoidStateType.Dead
end

local function isIgnored(p, pc)
	return config.ignoreTeam and isTeammate(p, pc)
end

local function setupPart(part, pData)
	if not part:IsA("BasePart") then return end
	local props = { Size = part.Size, Transparency = part.Transparency, Massless = part.Massless, CanCollide = part.CanCollide }
	local gs = part:AddGetHook("Size", props.Size)
	local gt = part:AddGetHook("Transparency", props.Transparency)
	local gm = part:AddGetHook("Massless", props.Massless)
	local gc = part:AddGetHook("CanCollide", props.CanCollide)

	local ss = part:AddSetHook("Size", function(_, v)
		props.Size = v; gs:Modify("Size", props.Size)
		if config.toggled and pData.alive and not pData.ignored and selectedParts[part.Name] then return Vector3.new(config.size, config.size, config.size) end
		return props.Size
	end)
	local st = part:AddSetHook("Transparency", function(_, v)
		props.Transparency = v; gt:Modify("Transparency", props.Transparency)
		if config.toggled and pData.alive and not pData.ignored and selectedParts[part.Name] then return config.transparency end
		return props.Transparency
	end)
	local sm = part:AddSetHook("Massless", function(_, v)
		props.Massless = v; gm:Modify("Massless", props.Massless)
		if config.toggled and part.Name ~= "HumanoidRootPart" and pData.alive and not pData.ignored and selectedParts[part.Name] then return true end
		return props.Massless
	end)
	local sc = part:AddSetHook("CanCollide", function(_, v)
		props.CanCollide = v; gc:Modify("CanCollide", props.CanCollide)
		if config.toggled and config.noCollide and pData.alive and not pData.ignored and selectedParts[part.Name] then return false end
		return props.CanCollide
	end)

	part.Destroying:Connect(function()
		gs:Remove(); gt:Remove(); gm:Remove(); gc:Remove()
		ss:Remove(); st:Remove(); sm:Remove(); sc:Remove()
	end)
end

local function addPlayer(p)
	if p == lp or players[p] then return end
	local pData = { alive = true, ignored = false }
	players[p] = pData

	local function onChar(char)
		if not char then return end
		pData.alive = true
		local hum = char:FindFirstChildWhichIsA("Humanoid")
		if hum then
			hum.StateChanged:Connect(function(_, ns) pData.alive = ns ~= Enum.HumanoidStateType.Dead end)
		end
		for _, part in pairs(char:GetChildren()) do setupPart(part, pData) end
		char.ChildAdded:Connect(function(child) task.wait() setupPart(child, pData) end)
	end

	if p.Character then onChar(p.Character) end
	p.CharacterAdded:Connect(function(char) task.wait(0.3) onChar(char) end)
	p.CharacterRemoving:Connect(function() pData.alive = false end)
	p:GetPropertyChangedSignal("Team"):Connect(function()
		pData.ignored = isIgnored(p, p.Character)
	end)
end

for _, p in pairs(Players:GetPlayers()) do addPlayer(p) end
Players.PlayerAdded:Connect(addPlayer)
Players.PlayerRemoving:Connect(function(p) players[p] = nil end)

-- ESP
local espDrawings = {}
RunService:BindToRenderStep("XiutESP", Enum.RenderPriority.Camera.Value - 1, function()
	if not config.espNames then
		for p, d in pairs(espDrawings) do d.Visible = false end
		return
	end
	Camera = Workspace.CurrentCamera
	if not Camera then return end

	for p, _ in pairs(players) do
		if not p.Character or not players[p] or not players[p].alive or (config.ignoreTeam and isTeammate(p, p.Character)) then
			if espDrawings[p] then espDrawings[p].Visible = false end
		else
			local root = p.Character:FindFirstChild("HumanoidRootPart") or p.Character:FindFirstChild("Torso")
			if root then
				if not espDrawings[p] then
					local d = Drawing.new("Text"); d.Center = true; d.Outline = true; d.Size = 16; d.OutlineColor = Color3.fromRGB(0, 0, 0)
					espDrawings[p] = d
				end
				local pos, vis = Camera:WorldToViewportPoint(root.Position)
				if vis then
					local dist = math.floor((Camera.CFrame.Position - root.Position).Magnitude)
					espDrawings[p].Color = p.Team == lp.Team and Color3.fromRGB(100, 255, 100) or Color3.fromRGB(255, 100, 100)
					espDrawings[p].Text = p.Name .. " [" .. dist .. "m]"
					espDrawings[p].Position = Vector2.new(pos.X, pos.Y - 50)
					espDrawings[p].Visible = true
				else
					espDrawings[p].Visible = false
				end
			elseif espDrawings[p] then
				espDrawings[p].Visible = false
			end
		end
	end
end)
