local function addSlider(text, min, max, defaultVal, callback)
	local frame = Instance.new("Frame")
	frame.Size = UDim2.new(1, 0, 0, 48)
	frame.BackgroundColor3 = theme.CardBg
	frame.Parent = contentArea
	addCorner(frame, 8)
	addStroke(frame, theme.Border, 1)

	local label = Instance.new("TextLabel")
	label.Size = UDim2.new(0.7, 0, 0, 20)
	label.Position = UDim2.fromOffset(10, 4)
	label.BackgroundTransparency = 1
	label.Text = text
	label.TextColor3 = theme.TextActive
	label.Font = Enum.Font.GothamMedium
	label.TextSize = 12
	label.TextXAlignment = Enum.TextXAlignment.Left
	label.Parent = frame

	local valueLabel = Instance.new("TextLabel")
	valueLabel.Size = UDim2.new(0.25, 0, 0, 20)
	valueLabel.Position = UDim2.new(0.7, 0, 0, 4)
	valueLabel.BackgroundTransparency = 1
	valueLabel.Text = tostring(defaultVal)
	valueLabel.TextColor3 = theme.AccentGlow
	valueLabel.Font = Enum.Font.GothamBold
	valueLabel.TextSize = 12
	valueLabel.TextXAlignment = Enum.TextXAlignment.Right
	valueLabel.TextScaled = true
	valueLabel.Parent = frame

	local sliderBg = Instance.new("Frame")
	sliderBg.Size = UDim2.new(1, -20, 0, 6)
	sliderBg.Position = UDim2.new(0, 10, 0.7, 0)
	sliderBg.BackgroundColor3 = Color3.fromRGB(50, 50, 60)
	sliderBg.BorderSizePixel = 0
	sliderBg.Parent = frame
	addCorner(sliderBg, 3)

	local sliderFill = Instance.new("Frame")
	sliderFill.Size = UDim2.fromScale((defaultVal - min) / (max - min), 1)
	sliderFill.BackgroundColor3 = theme.AccentGlow
	sliderFill.BorderSizePixel = 0
	sliderFill.Parent = sliderBg
	addCorner(sliderFill, 3)

	local sliderBtn = Instance.new("TextButton")
	sliderBtn.Size = UDim2.fromOffset(14, 14)
	sliderBtn.Position = UDim2.new((defaultVal - min) / (max - min), -7, 0.5, -7)
	sliderBtn.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
	sliderBtn.Text = ""
	sliderBtn.Parent = sliderBg
	addCorner(sliderBtn, 7)
	addStroke(sliderBtn, theme.AccentGlow, 1)

	local dragging = false
	local function updateSlider(input)
		local totalWidth = sliderBg.AbsoluteSize.X
		local inputX = input.Position.X - sliderBg.AbsolutePosition.X
		local ratio = math.clamp(inputX / totalWidth, 0, 1)
		sliderFill.Size = UDim2.fromScale(ratio, 1)
		sliderBtn.Position = UDim2.new(ratio, -7, 0.5, -7)
		local val = min + (max - min) * ratio
		valueLabel.Text = tostring(val)
		callback(val)
	end

	sliderBtn.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = true
		end
	end)
	UserInputService.InputChanged:Connect(function(input)
		if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
			updateSlider(input)
		end
	end)
	UserInputService.InputEnded:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = false
		end
	end)

	return { setValue = function(val)
		local ratio = (val - min) / (max - min)
		sliderFill.Size = UDim2.fromScale(ratio, 1)
		sliderBtn.Position = UDim2.new(ratio, -7, 0.5, -7)
		valueLabel.Text = tostring(val)
	end }
end

-- Draggable
local function makeDraggable(guiFrame)
	local dragging, dragInput, dragStart, startPosition
	local function update(input)
		local delta = input.Position - dragStart
		guiFrame.Position = UDim2.new(
			startPosition.X.Scale,
			startPosition.X.Offset + delta.X,
			startPosition.Y.Scale,
			startPosition.Y.Offset + delta.Y
		)
	end
	guiFrame.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = true
			dragStart = input.Position
			startPosition = guiFrame.Position
			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then
					dragging = false
				end
			end)
		end
	end)
	guiFrame.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
			dragInput = input
		end
	end)
	UserInputService.InputChanged:Connect(function(input)
		if input == dragInput and dragging then
			update(input)
		end
	end)
end
makeDraggable(mainFrame)

-- ==================== HITBOX EXTENDER LOGIC ====================

local hitboxEnabled = false
local hitboxSize = 8
local hitboxTransparency = 0.5
local noCollision = false
local selectedParts = {
	HumanoidRootPart = true,
	Head = true,
	Torso = true,
	["Left Arm"] = true,
	["Right Arm"] = true,
	["Left Leg"] = true,
	["Right Leg"] = true,
}

local playersData = {}
local Camera = Workspace.CurrentCamera

local function isPartSelected(partName)
	for name, enabled in pairs(selectedParts) do
		if enabled then
			if name == "Left Arm" and (string.match(partName, "Left") and (string.match(partName, "Arm") or string.match(partName, "Hand"))) then
				return true
			end
			if name == "Right Arm" and (string.match(partName, "Right") and (string.match(partName, "Arm") or string.match(partName, "Hand"))) then
				return true
			end
			if name == "Left Leg" and (string.match(partName, "Left") and (string.match(partName, "Leg") or string.match(partName, "Foot"))) then
				return true
			end
			if name == "Right Leg" and (string.match(partName, "Right") and (string.match(partName, "Leg") or string.match(partName, "Foot"))) then
				return true
			end
			if partName == name then
				return true
			end
		end
	end
	return false
end

local function setupPartHooks(part, data)
	if not part:IsA("BasePart") then return end

	local props = {
		Size = part.Size,
		Transparency = part.Transparency,
		Massless = part.Massless,
		CanCollide = part.CanCollide,
	}

	local getSizeHook = part:AddGetHook("Size", props.Size)
	local getTransparencyHook = part:AddGetHook("Transparency", props.Transparency)
	local getMasslessHook = part:AddGetHook("Massless", props.Massless)
	local getCanCollideHook = part:AddGetHook("CanCollide", props.CanCollide)

	local setSizeHook = part:AddSetHook("Size", function(_, value)
		props.Size = value
		getSizeHook:Modify("Size", props.Size)
		if hitboxEnabled and isPartSelected(part.Name) and data.isActive and not data.isDead then
			return Vector3.new(hitboxSize, hitboxSize, hitboxSize)
		end
		return props.Size
	end)

	local setTransparencyHook = part:AddSetHook("Transparency", function(_, value)
		props.Transparency = value
		getTransparencyHook:Modify("Transparency", props.Transparency)
		if hitboxEnabled and isPartSelected(part.Name) and data.isActive and not data.isDead then
			return hitboxTransparency
		end
		return props.Transparency
	end)

	local setMasslessHook = part:AddSetHook("Massless", function(_, value)
		props.Massless = value
		getMasslessHook:Modify("Massless", props.Massless)
		if hitboxEnabled and part.Name ~= "HumanoidRootPart" and data.isActive and not data.isDead then
			return true
		end
		return props.Massless
	end)

	local setCanCollideHook = part:AddSetHook("CanCollide", function(_, value)
		props.CanCollide = value
		getCanCollideHook:Modify("CanCollide", props.CanCollide)
		if hitboxEnabled and noCollision and data.isActive and not data.isDead then
			return false
		end
		return props.CanCollide
	end)

	part.Destroying:Connect(function()
		getSizeHook:Remove()
		getTransparencyHook:Remove()
		getMasslessHook:Remove()
		getCanCollideHook:Remove()
		setSizeHook:Remove()
		setTransparencyHook:Remove()
		setMasslessHook:Remove()
		setCanCollideHook:Remove()
	end)
end

local function setupCharacter(player, character)
	if not character then return end
	if not playersData[player] then
		playersData[player] = { isActive = true, isDead = false }
	end
	local data = playersData[player]

	local humanoid = character:FindFirstChildWhichIsA("Humanoid")
	if humanoid then
		humanoid.StateChanged:Connect(function(_, newState)
			data.isDead = newState == Enum.HumanoidStateType.Dead
		end)
	end

	for _, part in pairs(character:GetChildren()) do
		if part:IsA("BasePart") then
			setupPartHooks(part, data)
		end
	end

	character.ChildAdded:Connect(function(child)
		if child:IsA("BasePart") then
			setupPartHooks(child, data)
		end
	end)
end

local function addPlayer(player)
	if player == localPlayer then return end
	playersData[player] = { isActive = true, isDead = false }
	if player.Character then
		setupCharacter(player, player.Character)
	end
	player.CharacterAdded:Connect(function(character)
		setupCharacter(player, character)
	end)
end

for _, player in pairs(Players:GetPlayers()) do
	addPlayer(player)
end
Players.PlayerAdded:Connect(addPlayer)
Players.PlayerRemoving:Connect(function(player)
	playersData[player] = nil
end)

local function updateESP()
	if not espEnabled then
		for player, data in pairs(playersData) do
			if data.nameLabel then
				data.nameLabel.Visible = false
			end
		end
		return
	end
	local Camera = Workspace.CurrentCamera
	for player, data in pairs(playersData) do
		if not data.nameLabel then
			data.nameLabel = Drawing.new("Text")
			data.nameLabel.Center = true
			data.nameLabel.Outline = true
			data.nameLabel.Size = 16
			data.nameLabel.Color = Color3.fromRGB(255, 255, 255)
			data.nameLabel.OutlineColor = Color3.fromRGB(0, 0, 0)
		end
		local char = player.Character
		if char and char:FindFirstChild("HumanoidRootPart") then
			local root = char.HumanoidRootPart
			local pos, onScreen = Camera:WorldToViewportPoint(root.Position)
			if onScreen then
				data.nameLabel.Text = player.Name
				data.nameLabel.Position = Vector2.new(pos.X, pos.Y - 50)
				data.nameLabel.Visible = true
			else
				data.nameLabel.Visible = false
			end
		else
			data.nameLabel.Visible = false
		end
	end
end
