local function addCorner(p, r)
	local c = Instance.new("UICorner")

	c.Parent = p

local function addStroke(p, color, thick)
	local s = Instance.new("UIStroke")
	s.Color = color or theme.border
	s.Thickness = thick or 1
	s.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
	s.Parent = p
	return s
end

local function lerpColor(a, b, t)
	return Color3.new(a.R + (b.R - a.R) * t, a.G + (b.G - a.G) * t, a.B + (b.B - a.B) * t)
end

local function addSlider(text, min, max, default, onChange)
	local frame = Instance.new("Frame")
	frame.Size = UDim2.new(1, 0, 0, 52)
	frame.BackgroundColor3 = theme.card
	frame.Parent = content
	addCorner(frame, 8)
	addStroke(frame)

	local label = Instance.new("TextLabel")
	label.Size = UDim2.new(0.65, 0, 0, 20)
	label.Position = UDim2.fromOffset(12, 5)
	label.BackgroundTransparency = 1
	label.Text = text
	label.TextColor3 = theme.text
	label.Font = Enum.Font.GothamMedium
	label.TextSize = 12
	label.TextXAlignment = Enum.TextXAlignment.Left
	label.Parent = frame

	local valueLabel = Instance.new("TextLabel")
	valueLabel.Size = UDim2.new(0.3, 0, 0, 20)
	valueLabel.Position = UDim2.new(0.7, 0, 0, 5)
	valueLabel.BackgroundTransparency = 1
	valueLabel.Text = tostring(default)
	valueLabel.TextColor3 = theme.accent
	valueLabel.Font = Enum.Font.GothamBold
	valueLabel.TextSize = 13
	valueLabel.TextXAlignment = Enum.TextXAlignment.Right
	valueLabel.TextScaled = true
	valueLabel.Parent = frame

	local track = Instance.new("Frame")
	track.Size = UDim2.new(1, -24, 0, 6)
	track.Position = UDim2.new(0, 12, 0, 32)
	track.BackgroundColor3 = Color3.fromRGB(50, 50, 60)
	track.BorderSizePixel = 0
	track.Parent = frame
	addCorner(track, 3)

	local fill = Instance.new("Frame")
	fill.Size = UDim2.fromScale((default - min) / (max - min), 1)
	fill.BackgroundColor3 = theme.accent
	fill.BorderSizePixel = 0
	fill.Parent = track
	addCorner(fill, 3)

	local thumb = Instance.new("TextButton")
	thumb.Size = UDim2.fromOffset(16, 16)
	thumb.Position = UDim2.new((default - min) / (max - min), -8, 0.5, -8)
	thumb.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
	thumb.Text = ""
	thumb.Parent = track
	addCorner(thumb, 8)
	addStroke(thumb, theme.accent, 1.5)

	local dragging = false
	local function update(inp)
		local w = track.AbsoluteSize.X
		if w <= 0 then return end
		local ratio = math.clamp((inp.Position.X - track.AbsolutePosition.X) / w, 0, 1)
		fill.Size = UDim2.fromScale(ratio, 1)
		thumb.Position = UDim2.new(ratio, -8, 0.5, -8)
		local val = min + (max - min) * ratio
		valueLabel.Text = tostring(math.floor(val * 100) / 100)
		onChange(val)
	end

	thumb.InputBegan:Connect(function(inp)
		if inp.UserInputType == Enum.UserInputType.MouseButton1 or inp.UserInputType == Enum.UserInputType.Touch then
			dragging = true
		end
	end)

	local conn1 = UserInputService.InputChanged:Connect(function(inp)
		if dragging and (inp.UserInputType == Enum.UserInputType.MouseMovement or inp.UserInputType == Enum.UserInputType.Touch) then
			update(inp)
		end
	end)

	local conn2 = UserInputService.InputEnded:Connect(function(inp)
		if inp.UserInputType == Enum.UserInputType.MouseButton1 or inp.UserInputType == Enum.UserInputType.Touch then
			dragging = false
		end
	end)

	frame.Destroying:Connect(function()
		conn1:Disconnect()
		conn2:Disconnect()
	end)

	return { set = function(val)
		local ratio = (val - min) / (max - min)
		fill.Size = UDim2.fromScale(ratio, 1)
		thumb.Position = UDim2.new(ratio, -8, 0.5, -8)
		valueLabel.Text = tostring(math.floor(val * 100) / 100)
	end }
end

-- Draggable
local function makeDraggable(g)
	local drag, inp, start, pos
	local function update(inp)
		local d = inp.Position - start
		g.Position = UDim2.new(pos.X.Scale, pos.X.Offset + d.X, pos.Y.Scale, pos.Y.Offset + d.Y)
	end
	g.InputBegan:Connect(function(i)
		if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then
			drag = true; start = i.Position; pos = g.Position
			i.Changed:Connect(function() if i.UserInputState == Enum.UserInputState.End then drag = false end end)
		end
	end)
	g.InputChanged:Connect(function(i)
		if i.UserInputType == Enum.UserInputType.MouseMovement or i.UserInputType == Enum.UserInputType.Touch then inp = i end
	end)
	UserInputService.InputChanged:Connect(function(i)
		if i == inp and drag then update(i) end
	end)
end

local config = {
	enabled = false,
	size = 5,
	transparency = 0.4,
	noCollide = false,
	parts = {
		Head = true,
		HumanoidRootPart = true,
		Torso = true,
		["Left Arm"] = true,
		["Right Arm"] = true,
		["Left Leg"] = true,
		["Right Leg"] = true,
	},
	espEnabled = false,
	espNames = false,
	ignoreTeam = true,
}

local playerData = {}
local espNameObjects = {}

local function isPartMatch(name, cfgName)
	if cfgName == "Left Arm" then return string.match(name, "Left") and (string.match(name, "Arm") or string.match(name, "Hand")) end
	if cfgName == "Right Arm" then return string.match(name, "Right") and (string.match(name, "Arm") or string.match(name, "Hand")) end
	if cfgName == "Left Leg" then return string.match(name, "Left") and (string.match(name, "Leg") or string.match(name, "Foot")) end
	if cfgName == "Right Leg" then return string.match(name, "Right") and (string.match(name, "Leg") or string.match(name, "Foot")) end
	return name == cfgName
end

local function hookPart(part, pData)
	if not part:IsA("BasePart") then return end

	local props = {
		Size = part.Size,
		Transparency = part.Transparency,
		Massless = part.Massless,
		CanCollide = part.CanCollide,
	}

	local gSize = part:AddGetHook("Size", props.Size)
	local gTrans = part:AddGetHook("Transparency", props.Transparency)
	local gMass = part:AddGetHook("Massless", props.Massless)
	local gColl = part:AddGetHook("CanCollide", props.CanCollide)

	local sSize = part:AddSetHook("Size", function(_, v)
		props.Size = v
		gSize:Modify("Size", props.Size)
		if config.enabled and pData.alive then
			for name, active in pairs(config.parts) do
				if active and isPartMatch(part.Name, name) then
					return Vector3.new(config.size, config.size, config.size)
				end
			end
		end
		return props.Size
	end)

	local sTrans = part:AddSetHook("Transparency", function(_, v)
		props.Transparency = v
		gTrans:Modify("Transparency", props.Transparency)
		if config.enabled and pData.alive then
			for name, active in pairs(config.parts) do
				if active and isPartMatch(part.Name, name) then
					return config.transparency
				end
			end
		end
		return props.Transparency
	end)

	local sMass = part:AddSetHook("Massless", function(_, v)
		props.Massless = v
		gMass:Modify("Massless", props.Massless)
		if config.enabled and part.Name ~= "HumanoidRootPart" and pData.alive then
			for name, active in pairs(config.parts) do
				if active and isPartMatch(part.Name, name) then
					return true
				end
			end
		end
		return props.Massless
	end)

	local sColl = part:AddSetHook("CanCollide", function(_, v)
		props.CanCollide = v
		gColl:Modify("CanCollide", props.CanCollide)
		if config.enabled and config.noCollide and pData.alive then
			for name, active in pairs(config.parts) do
				if active and isPartMatch(part.Name, name) then
					return false
				end
			end
		end
		return props.CanCollide
	end)

	local cleanup = {}
	table.insert(cleanup, part.Destroying:Connect(function()
		gSize:Remove(); gTrans:Remove(); gMass:Remove(); gColl:Remove()
		sSize:Remove(); sTrans:Remove(); sMass:Remove(); sColl:Remove()
		for _, c in pairs(cleanup) do c:Disconnect() end
	end))
end

local function setupPlayer(player)
	if player == localPlayer then return end
	if playerData[player] then return end

	local pData = { alive = true }
	playerData[player] = pData

	local function onChar(char)
		if not char then return end
		pData.alive = true

		local hum = char:FindFirstChildWhichIsA("Humanoid")
		if hum then
			hum.StateChanged:Connect(function(_, newState)
				pData.alive = newState ~= Enum.HumanoidStateType.Dead
			end)
			hum.Died:Connect(function()
				pData.alive = false
			end)
		end

		for _, part in pairs(char:GetChildren()) do
			hookPart(part, pData)
		end

		char.ChildAdded:Connect(function(child)
			task.wait()
			hookPart(child, pData)
		end)
	end

	if player.Character then
		onChar(player.Character)
	end

	player.CharacterAdded:Connect(function(char)
		task.wait(0.3)
		onChar(char)
	end)

	player.CharacterRemoving:Connect(function()
		pData.alive = false
	end)
end

for _, p in pairs(Players:GetPlayers()) do setupPlayer(p) end
Players.PlayerAdded:Connect(setupPlayer)
Players.PlayerRemoving:Connect(function(p)
	playerData[p] = nil
	if espNameObjects[p] then
		espNameObjects[p]:Remove()
		espNameObjects[p] = nil
	end
end)

-- ESP System
local function updateESP()
	if not config.espNames then
		for p, obj in pairs(espNameObjects) do
			obj.Visible = false
		end
		return
	end

	Camera = Workspace.CurrentCamera
	if not Camera then return end

	for _, p in pairs(Players:GetPlayers()) do
		if p == localPlayer then
			if espNameObjects[p] then
				espNameObjects[p].Visible = false
			end
			continue
		end

		if config.ignoreTeam and p.Team == localPlayer.Team then
			if espNameObjects[p] then espNameObjects[p].Visible = false end
			continue
		end

		local char = p.Character
		if not char or not pData[p] or not pData[p].alive then
			if espNameObjects[p] then espNameObjects[p].Visible = false end
			continue
		end

		local root = char:FindFirstChild("HumanoidRootPart") or char:FindFirstChild("Torso")
		if not root then
			if espNameObjects[p] then espNameObjects[p].Visible = false end
			continue
		end

		if not espNameObjects[p] then
			local obj = Drawing.new("Text")
			obj.Center = true
			obj.Outline = true
			obj.Size = 18
			obj.Color = Color3.fromRGB(255, 255, 255)
			obj.OutlineColor = Color3.fromRGB(0, 0, 0)
			espNameObjects[p] = obj
		end

		local pos, vis = Camera:WorldToViewportPoint(root.Position)
		if vis then
			local dist = math.floor((Camera.CFrame.Position - root.Position).Magnitude)
			local col = p.Team == localPlayer.Team and Color3.fromRGB(100, 255, 100) or Color3.fromRGB(255, 100, 100)
			espNameObjects[p].Color = col
			espNameObjects[p].Text = p.Name .. " [" .. dist .. "m]"
			espNameObjects[p].Position = Vector2.new(pos.X, pos.Y - 55)
			espNameObjects[p].Size = math.clamp(400 / math.max(pos.Z, 1), 10, 40)
			espNameObjects[p].Visible = true
		else
			espNameObjects[p].Visible = false
		end
	end
end

RunService:BindToRenderStep("XiutESP", Enum.RenderPriority.Camera.Value - 1, updateESP)
