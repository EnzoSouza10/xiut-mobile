local ok, err = pcall(function()

if getgenv().XitLoaded then
	return
end
getgenv().XitLoaded = true

if not game:IsLoaded() then
	game.Loaded:Wait()
end

local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local Workspace = game:GetService("Workspace")
local UserInputService = game:GetService("UserInputService")
local Camera = Workspace.CurrentCamera
local lPlayer = Players.LocalPlayer
local players = {}
local enabled = false
local hitboxSize = 10
local hitboxTransp = 0.5

-- UI
local gui = Instance.new("ScreenGui")
gui.Name = "XitGUI"
gui.ResetOnSpawn = false
gui.Parent = game:GetService("CoreGui")

local frame = Instance.new("Frame")
frame.Size = UDim2.new(0, 250, 0, 180)
frame.Position = UDim2.new(0.5, -125, 0.5, -90)
frame.BackgroundColor3 = Color3.fromRGB(20, 20, 30)
frame.BorderSizePixel = 0
frame.Active = true
frame.Draggable = true
frame.Parent = gui

local title = Instance.new("TextLabel")
title.Size = UDim2.new(1, 0, 0, 30)
title.BackgroundColor3 = Color3.fromRGB(40, 40, 60)
title.TextColor3 = Color3.fromRGB(255, 255, 255)
title.Text = "XIT - Hitbox Extender"
title.Font = Enum.Font.GothamBold
title.TextSize = 16
title.Parent = frame

local toggleBtn = Instance.new("TextButton")
toggleBtn.Size = UDim2.new(0, 200, 0, 35)
toggleBtn.Position = UDim2.new(0.5, -100, 0, 40)
toggleBtn.BackgroundColor3 = Color3.fromRGB(60, 180, 60)
toggleBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
toggleBtn.Text = "Enabled"
toggleBtn.Font = Enum.Font.Gotham
toggleBtn.TextSize = 14
toggleBtn.Parent = frame

local sizeSlider = Instance.new("TextBox")
sizeSlider.Size = UDim2.new(0, 200, 0, 30)
sizeSlider.Position = UDim2.new(0.5, -100, 0, 85)
sizeSlider.BackgroundColor3 = Color3.fromRGB(40, 40, 50)
sizeSlider.TextColor3 = Color3.fromRGB(255, 255, 255)
sizeSlider.PlaceholderText = "Size (2-50): " .. hitboxSize
sizeSlider.Text = ""
sizeSlider.Font = Enum.Font.Gotham
sizeSlider.TextSize = 14
sizeSlider.ClearTextOnFocus = false
sizeSlider.Parent = frame

local transpSlider = Instance.new("TextBox")
transpSlider.Size = UDim2.new(0, 200, 0, 30)
transpSlider.Position = UDim2.new(0.5, -100, 0, 122)
transpSlider.BackgroundColor3 = Color3.fromRGB(40, 40, 50)
transpSlider.TextColor3 = Color3.fromRGB(255, 255, 255)
transpSlider.PlaceholderText = "Transparency (0-1): " .. hitboxTransp
transpSlider.Text = ""
transpSlider.Font = Enum.Font.Gotham
transpSlider.TextSize = 14
transpSlider.ClearTextOnFocus = false
transpSlider.Parent = frame

toggleBtn.MouseButton1Click:Connect(function()
	enabled = not enabled
	toggleBtn.BackgroundColor3 = enabled and Color3.fromRGB(60, 180, 60) or Color3.fromRGB(180, 60, 60)
	toggleBtn.Text = enabled and "Enabled" or "Disabled"
	updatePlayers()
end)

sizeSlider.FocusLost:Connect(function()
	local val = tonumber(sizeSlider.Text)
	if val then
		hitboxSize = math.clamp(val, 2, 50)
	end
	sizeSlider.PlaceholderText = "Size (2-50): " .. hitboxSize
	sizeSlider.Text = ""
	updatePlayers()
end)

transpSlider.FocusLost:Connect(function()
	local val = tonumber(transpSlider.Text)
	if val then
		hitboxTransp = math.clamp(val, 0, 1)
	end
	transpSlider.PlaceholderText = "Transparency (0-1): " .. hitboxTransp
	transpSlider.Text = ""
	updatePlayers()
end)

-- toggle UI com Insert
UserInputService.InputBegan:Connect(function(input, gpe)
	if gpe then return end
	if input.KeyCode == Enum.KeyCode.Insert then
		gui.Enabled = not gui.Enabled
	end
end)

-- Hitbox extender

local function updatePlayers()
	for _, idx in pairs(players) do
		task.spawn(function()
			idx:Update()
		end)
	end
end

local function addPlayer(player)
	if player == lPlayer then return end
	players[player] = {}
	local idx = players[player]
	local char = player.Character

	local defaultProps = {}

	local function resize(part)
		if not defaultProps[part.Name] then
			defaultProps[part.Name] = {
				Size = part.Size,
				Transparency = part.Transparency,
				Massless = part.Massless,
				CanCollide = part.CanCollide
			}
		end
		if enabled then
			part.Size = Vector3.new(hitboxSize, hitboxSize, hitboxSize)
			part.Transparency = hitboxTransp
		else
			part.Size = defaultProps[part.Name].Size
			part.Transparency = defaultProps[part.Name].Transparency
		end
	end

	function idx:Update()
		if not char then return end
		for _, v in pairs(char:GetChildren()) do
			if v:IsA("BasePart") then
				resize(v)
			end
		end
	end

	if char then
		idx:Update()
	end

	player.CharacterAdded:Connect(function(c)
		char = c
		defaultProps = {}
		task.wait(0.5)
		idx:Update()
	end)

	player.CharacterRemoving:Connect(function()
		char = nil
		defaultProps = {}
	end)
end

for _, p in ipairs(Players:GetPlayers()) do
	addPlayer(p)
end

Players.PlayerAdded:Connect(addPlayer)

end) -- pcall

if not ok then
	warn("XIT Error:", err)
	local gui = Instance.new("ScreenGui")
	gui.Name = "XitError"
	gui.ResetOnSpawn = false
	gui.Parent = game:GetService("CoreGui")
	local f = Instance.new("Frame")
	f.Size = UDim2.new(0, 400, 0, 150)
	f.Position = UDim2.new(0.5, -200, 0.5, -75)
	f.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
	f.BorderSizePixel = 0
	f.Parent = gui
	local t = Instance.new("TextLabel")
	t.Size = UDim2.new(1, -20, 1, -20)
	t.Position = UDim2.new(0, 10, 0, 10)
	t.BackgroundTransparency = 1
	t.TextColor3 = Color3.fromRGB(255, 100, 100)
	t.TextScaled = true
	t.TextWrapped = true
	t.Text = "XIT Error:\n" .. tostring(err)
	t.Font = Enum.Font.Code
	t.Parent = f
	local b = Instance.new("TextButton")
	b.Size = UDim2.new(0, 80, 0, 30)
	b.Position = UDim2.new(0.5, -40, 1, -40)
	b.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
	b.TextColor3 = Color3.fromRGB(255, 255, 255)
	b.Text = "OK"
	b.Parent = f
	b.MouseButton1Click:Connect(function() gui:Destroy() end)
end
