local function isTeammate(lPlayer, player, playerChar)
	if game.GameId == 718936923 then
		if not lPlayer.Character or not playerChar or not playerChar:FindFirstChild("HumanoidRootPart") then return true end
		return lPlayer.Character.HumanoidRootPart.Color == playerChar.HumanoidRootPart.Color
	elseif game.PlaceId == 633284182 then
		if not player:FindFirstChild("PlayerData") or not player.PlayerData:FindFirstChild("TeamValue") then return true end
		return lPlayer.PlayerData.TeamValue.Value == player.PlayerData.TeamValue.Value
	elseif game.PlaceId == 2029250188 then
		if not lPlayer.Character or not playerChar then return true end
		return lPlayer.Character.Parent == playerChar.Parent
	elseif game.PlaceId == 2978450615 then
		local s, d = pcall(function() return getrenv()._G.PlayerProfiles.Data end)
		if not s or not d then return lPlayer.Team == player.Team end
		return d[lPlayer.Name].Team == d[player.Name].Team
	elseif game.GameId == 1934496708 then
		if W:FindFirstChild("FriendlyFire") and W.FriendlyFire.Value then return false end
		if not player.Team or player.Team.Name == "LOBBY" or lPlayer.Team.Name == "LOBBY" or player.Team.Name == "Admin" then return true end
		if lPlayer.Team == player.Team then return true end
		if teamModule then
			local lt, pt = teamModule[lPlayer.Team.Name], teamModule[player.Team.Name]
			return lt == pt or (lt == "CI" and pt == "CD") or (pt == "CI" and lt == "CD")
		end
		return false
	elseif game.PlaceId == 2622527242 then
		if not player.Team or player.Team.Name == "Intro" or player.Team.Name == "Spectator" or player.Team.Name == "Not Playing" then return true end
		if lPlayer.Team == player.Team then return true end
		local function mapTeam(n)
			if n == "Class-D Personnel" or n == "Chaos Insurgency" then return "A" end
			if n == "Facility Personnel" or n == "Security Department" or n == "Mobile Task Force" then return "B" end
			if n == "SCPs" or n == "Serpent's Hand" then return "C" end
			if n == "Global Occult Coalition" then return "D" end
			if n == "Unusual Incidents Unit" then return "E" end
			return "F"
		end
		local lt, pt = mapTeam(lPlayer.Team.Name), mapTeam(player.Team.Name)
		if lt == "E" or pt == "E" then return lt == "B" or pt == "B" or lt == "D" or pt == "D" end
		return lt == pt
	elseif game.PlaceId == 8770868695 then
		if not lPlayer.Character or not playerChar or not player.Team then return true end
		if player.Team.Name == "Dead" or player.Team.Name == "Inactive" then return true end
		return lPlayer.Character.Parent == playerChar.Parent
	elseif game.PlaceId == 5884786982 then
		if not lPlayer.Character or not playerChar then return true end
		return lPlayer.Character.Name ~= "Killer" and playerChar.Name ~= "Killer"
	elseif game.GameId == 2162282815 then
		if not player:FindFirstChild("SelectedTeam") then return true end
		return player.SelectedTeam.Value == lPlayer.SelectedTeam.Value
	elseif game.PlaceId == 1240644540 then
		if not teamModule or not teamModule.IsPlayerSurvivor then return true end
		return teamModule.IsPlayerSurvivor(nil, player) and teamModule.IsPlayerSurvivor(nil, lPlayer)

		if not player:FindFirstChild("PlayerData") or not player.PlayerData:FindFirstChild("Team") then return true end
		return lPlayer.PlayerData.Team.Value == player.PlayerData.Team.Value

local function isDead(playerChar)
	if not playerChar then return true end
	local humanoid = playerChar:FindFirstChildWhichIsA("Humanoid")
	if game.PlaceId == 6172932937 then
		local pl = playerChar:FindFirstAncestorOfClass("Player")
		if pl and pl:FindFirstChild("ragdolled") then return pl.ragdolled.Value end
	end
	if game.GameId == 718936923 then return playerChar:FindFirstChild("Dead") ~= nil end
	return humanoid and humanoid:GetState() == Enum.HumanoidStateType.Dead
end

local function isSitting(playerChar)
	local h = playerChar and playerChar:FindFirstChildWhichIsA("Humanoid")
	return cfg.si and h ~= nil and h.Sit == true
end

local function isFFed(playerChar)
	if not playerChar then return false end
	if game.PlaceId == 4991214437 or game.PlaceId == 6652350934 then
		local head = playerChar:FindFirstChild("Head")
		return head and head.Material == Enum.Material.ForceField
	end
	local ff = playerChar:FindFirstChildWhichIsA("ForceField")
	return cfg.ff and ff ~= nil and ff.Visible == true
end

local function shouldIgnore(player, playerChar)
	if not playerChar then return true end
	if isDead(playerChar) then return true end
	if isSitting(playerChar) then return true end
	if isFFed(playerChar) then return true end
	if ignorePlayers[player] then return true end
	if ignoreTeams[tostring(player.Team)] then return true end
	return false
end

local function sv(c)
	if not c then return end; df[c] = {}
	for _,p in pairs(c:GetChildren()) do
		if p:IsA("BasePart") then df[c][p] = {Size=p.Size,T=p.Transparency,CC=p.CanCollide,ML=p.Massless} end
	end
end
local function rs(char)
	if not char or not df[char] then return end
	for p,pr in pairs(df[char]) do
		if p and p.Parent then p.Size=pr.Size;p.Transparency=pr.T;p.CanCollide=pr.CC;p.Massless=pr.ML end
	end
end

local function isPartSelected(name)
	if sel["HumanoidRootPart"] and name == "HumanoidRootPart" then return true end
	if sel["Head"] and (name == "Head" or name:find("Head")) then return true end
	if sel["Torso"] and (name:find("Torso") or name:find("Torso")) then return true end
	if sel["Left Arm"] and name:find("Left") and (name:find("Arm") or name:find("Hand") or name:find("Forearm") or name:find("Shoulder")) then return true end
	if sel["Right Arm"] and name:find("Right") and (name:find("Arm") or name:find("Hand") or name:find("Forearm") or name:find("Shoulder")) then return true end
	if sel["Left Leg"] and name:find("Left") and (name:find("Leg") or name:find("Foot") or name:find("Thigh") or name:find("Calf")) then return true end
	if sel["Right Leg"] and name:find("Right") and (name:find("Leg") or name:find("Foot") or name:find("Thigh") or name:find("Calf")) then return true end
	if sel["Custom Part"] and name:find(cfg.customPart) then return true end
	return false
end

local function ex(char, plr)
	if not char then return end
	if plr and shouldIgnore(plr, char) then rs(char); return end
	for _,p in pairs(char:GetChildren()) do
		if p:IsA("BasePart") and isPartSelected(p.Name) then
			p.Size = Vector3.new(cfg.sz,cfg.sz,cfg.sz)
			p.Transparency = cfg.tr
			if not cfg.nc then p.CanCollide = false end
			if p.Name ~= "HumanoidRootPart" then p.Massless = true end
		end
	end
end

local function ad()
	if not cfg.ad or not cfg.en or adActive then return end
	adActive = true
	for p,pl in pairs(pd) do
		if pl.al and p.Character and not shouldIgnore(p, p.Character) then rs(p.Character) end
	end
	spawn(function() wf(cfg.adD) adActive = false end)
end

local function ap(p)
	if p == lp then return end
	local pl = {al=true,ch=nil,espName=nil,chams=nil}
	pd[p] = pl
	local function oc(char)
		if not char then return end
		pl.al = true; pl.ch = char
		local h = char:FindFirstChildWhichIsA("Humanoid")
		if h then h.Died:Connect(function() pl.al = false end) end
		local root = char:WaitForChild("HumanoidRootPart", 5)
		if not root then root = char:WaitForChild("Torso", 3) end
		if not root then return end
		sv(char)
		if cfg.en then ex(char, p) end
		char.ChildAdded:Connect(function(ch) wf() if ch:IsA("BasePart") and cfg.en then ex(char, p) end end)
		char.ChildRemoved:Connect(function(ch) if ch:IsA("BasePart") then sv(char) end end)
	end
	if p.Character then oc(p.Character) end
	p.CharacterAdded:Connect(function(char) wf(0.5) oc(char) end)
	p.CharacterRemoving:Connect(function() pl.al = false end)
end
for _,p in pairs(P:GetPlayers()) do ap(p) end
P.PlayerAdded:Connect(ap)
P.PlayerRemoving:Connect(function(p)
	if pd[p] then
		if pd[p].espName then pcall(function() pd[p].espName:Remove() end) end
		if pd[p].chams then pcall(function() pd[p].chams:Destroy() end) end
	end
	pd[p] = nil
end)

	for p,pl in pairs(pd) do
		if p.Character then
			sv(p.Character)
			if cfg.en then ex(p.Character, p) else rs(p.Character) end
		end
	end
end

				if pl.chams then pl.chams.Enabled = false end
			else
				-- name esp
				if cfg.es then
					if not pl.espName then
						local s,d = pcall(function()
							local d = Drawing.new("Text"); d.Center = true; d.Outline = true; d.Size = 14; d.OutlineColor = Color3.fromRGB(0,0,0); return d
						end)
						if s then pl.espName = d end
					end
					if pl.espName then
						local pos, vis = C:WorldToViewportPoint(rt.Position)
						if vis then
							pl.espName.Color = cfg.eu and p.TeamColor.Color or Color3.fromRGB(255,255,255)
							pl.espName.Text = p.Name
							pl.espName.Position = Vector2.new(pos.X, pos.Y - 45)
							pl.espName.Visible = true
						else
							pl.espName.Visible = false
						end
					end
				elseif pl.espName then
					pl.espName.Visible = false
				end
				-- chams
				if cfg.ec then
					if not pl.chams then
						local s,c = pcall(function() local c = Instance.new("Highlight"); c.Parent = g; return c end)
						if s then pl.chams = c end
					end
					if pl.chams then
						pl.chams.Adornee = ch
						pl.chams.FillColor = cfg.eu and p.TeamColor.Color or Color3.fromRGB(0,180,255)
						pl.chams.OutlineColor = Color3.fromRGB(0,0,0)
						pl.chams.FillTransparency = 0.5
						pl.chams.OutlineTransparency = 0
						pl.chams.Enabled = true
					end
				elseif pl.chams then
					pl.chams.Enabled = false
				end
			end
		end
	end
end)
